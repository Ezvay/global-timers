<script>
const socket = io();
let activeSection = 'top';

// VOLUME
let volume = parseFloat(localStorage.getItem('volume')) || 0.2;
const volSlider = document.getElementById('volumeSlider');
volSlider.value = volume;
volSlider.oninput = () => { volume = parseFloat(volSlider.value); localStorage.setItem('volume', volume); };

// AUDIO
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
function beep(section, duration=0.15, freq=800) {
  if (section !== activeSection) return;
  const osc = audioCtx.createOscillator();
  const gain = audioCtx.createGain();
  osc.connect(gain); gain.connect(audioCtx.destination);
  osc.frequency.value = freq; osc.start();
  gain.gain.setValueAtTime(volume, audioCtx.currentTime);
  gain.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + duration);
  osc.stop(audioCtx.currentTime + duration);
}
function tripleBeep(s) { beep(s); setTimeout(()=>beep(s),250); setTimeout(()=>beep(s),500); }
function singleBeep(s) { beep(s, 0.7, 600); }

// SECTIONS
function activateSection(s) {
  activeSection = s;
  document.getElementById('sectionTop').classList.toggle('active', s==='top');
  document.getElementById('sectionBottom').classList.toggle('active', s==='bottom');
}
function toggleSection(s) {
  const grid = s==='top' ? document.getElementById('timersTop') : document.getElementById('timersBottom');
  grid.style.display = grid.style.display === 'none' ? 'grid' : 'none';
}

// LAYOUT
const layoutMap = { top: 'layout-2x4-horizontal', bottom: 'layout-2x4-horizontal' };
function setLayout(section, layout) {
  const grid = section==='top' ? document.getElementById('timersTop') : document.getElementById('timersBottom');
  grid.className = 'timer-grid ' + layout;
  layoutMap[section] = layout;
  localStorage.setItem('layout_' + section, layout);
  highlightLayoutBtn(section, layout);
}
function highlightLayoutBtn(section, layout) {
  const bar = document.getElementById('layoutBar' + (section==='top'?'Top':'Bottom'));
  const layouts = ['layout-2x4-horizontal','layout-2x4-vertical','layout-1x8-horizontal','layout-1x8-vertical'];
  [...bar.children].forEach((el, i) => el.classList.toggle('active-layout', layouts[i] === layout));
}
['top','bottom'].forEach(s => {
  const saved = localStorage.getItem('layout_' + s);
  if (saved) setLayout(s, saved);
  else highlightLayoutBtn(s, layoutMap[s]);
});

// FORMAT
function format(sec) {
  return String(Math.floor(sec/3600)).padStart(2,'0')+':'+
         String(Math.floor((sec%3600)/60)).padStart(2,'0')+':'+
         String(sec%60).padStart(2,'0');
}

// CREATE TIMER
function createTimer(id, section) {
  const card = document.createElement('div');
  card.className = 'timer-card state-grey';

  const name = document.createElement('div');
  name.className = 'timer-name';
  name.innerText = id.replace('KG_','').replace('KGD_','');

  const disp = document.createElement('div');
  disp.className = 'timer-display';
  disp.innerText = '00:00:00';

  const btns = document.createElement('div');
  btns.className = 'timer-btns';

  const btnStart = document.createElement('button');
  btnStart.className = 'btn btn-start'; btnStart.innerText = 'Start';
  const btnStop = document.createElement('button');
  btnStop.className = 'btn btn-stop'; btnStop.innerText = 'Stop';
  const btnReset = document.createElement('button');
  btnReset.className = 'btn btn-reset'; btnReset.innerText = 'Reset';

  const comment = document.createElement('input');
  comment.className = 'comment-input'; comment.placeholder = 'Notatka...';

  btnStart.onclick = () => { if (activeSection !== section) return; socket.emit('start', id); };
  btnStop.onclick  = () => { if (activeSection !== section) return; socket.emit('stop', id); };
  btnReset.onclick = () => { if (activeSection !== section) return; comment.value = ''; socket.emit('reset', id); };

  btns.append(btnStart, btnStop, btnReset);
  card.append(name, disp, btns, comment);
  return { card, disp, section, beepYellow: false, beepGreen: false };
}

const displays = {};
for (let i=1; i<=8; i++) {
  const id = 'KG_CH'+i;
  const t = createTimer(id, 'top');
  document.getElementById('timersTop').appendChild(t.card);
  displays[id] = t;
}
for (let i=1; i<=8; i++) {
  const id = 'KGD_CH'+i;
  const t = createTimer(id, 'bottom');
  document.getElementById('timersBottom').appendChild(t.card);
  displays[id] = t;
}

socket.on('update', data => {
  for (const id in data) {
    if (!displays[id]) continue;
    const sec = data[id];
    const t = displays[id];
    t.disp.innerText = format(sec);
    t.card.className = 'timer-card ' + (sec===0?'state-grey':sec<1500?'state-red':sec<1800?'state-yellow':'state-green');
    if (sec>=1500 && sec<1502 && !t.beepYellow) { tripleBeep(t.section); t.beepYellow = true; }
    if (sec>=1800 && sec<1802 && !t.beepGreen)  { singleBeep(t.section); t.beepGreen = true; }
    if (sec === 0) { t.beepYellow = false; t.beepGreen = false; }
  }
});
</script>
